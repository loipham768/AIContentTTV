    Create `components/editor/GrapesEditor.tsx`:

    ```
    'use client'

    import 'grapesjs/dist/css/grapes.min.css'
    import { GjsEditor } from '@grapesjs/react'   // adjust import based on actual package export
    import type { Editor } from 'grapesjs'
    import { MOCK_BLOCK } from '@/lib/mockBlock'

    interface GrapesEditorProps {
      onEditor: (editor: Editor) => void
    }

    export default function GrapesEditor({ onEditor }: GrapesEditorProps) {
      function handleEditor(editor: Editor) {
        editor.loadProjectData(MOCK_BLOCK)
        onEditor(editor)
      }

      return (
        <div className="flex-1 overflow-hidden relative">
          {/* Hidden containers for GrapesJS panel managers (keeps them out of visible UI) */}
          <div id="gjs-hidden-blocks" style={{ display: 'none' }} />
          <div id="gjs-hidden-layers" style={{ display: 'none' }} />
          <GjsEditor
            className="h-full"
            options={{
              height: '100%',
              storageManager: false,
              blockManager: { appendTo: '#gjs-hidden-blocks' },
              layerManager: { appendTo: '#gjs-hidden-layers' },
              styleManager: false,
              panels: { defaults: [] },
              deviceManager: {
                devices: [
                  { id: 'desktop', name: 'Desktop', width: '' },
                  { id: 'mobile', name: 'Mobile', width: '390px', widthMedia: '480px' },
                ],
              },
            }}
            onEditor={handleEditor}
          />
        </div>
      )
    }
    ```

    Important implementation notes:
    1. Check the actual export from @grapesjs/react. Look at `node_modules/@grapesjs/react/dist/index.d.ts` or `package.json` main field to find the real component export. The component might be `GjsEditor`, `Editor`, or a default export — adjust the import accordingly.
    2. The CSS import `'grapesjs/dist/css/grapes.min.css'` is correct for GrapesJS 0.22. Verify the path exists in `node_modules/grapesjs/dist/`. If the file is at a different path (e.g., `grapes.min.css` is in a different subdirectory), update accordingly.
    3. `loadProjectData(MOCK_BLOCK)` — the TypeScript type of MOCK_BLOCK (from `as const`) may not exactly match the expected parameter type. If TypeScript complains, cast: `editor.loadProjectData(MOCK_BLOCK as Parameters<typeof editor.loadProjectData>[0])`.
    4. The `styleManager: false` option — if GrapesJS 0.22 doesn't accept `false` for this option, use `styleManager: { sectors: [] }` as fallback.
    5. Do NOT add `export const runtime = 'nodejs'` — this is a client component, not a route handler.
    6. The hidden divs for block/layer managers must be rendered OUTSIDE the GjsEditor component to avoid GrapesJS trying to append to an element inside its own shadow DOM.

    After creating the file, run `npm run build` to catch TypeScript errors. Fix any type errors related to GrapesJS API before proceeding. Do not mark this task done until the build is clean.
