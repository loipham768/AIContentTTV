    Create `components/editor/EditorClientWrapper.tsx`:

    ```typescript
    'use client'

    import { useRef } from 'react'
    import dynamic from 'next/dynamic'
    import type { Editor } from 'grapesjs'
    import TopBar from '@/components/editor/TopBar'
    import PromptPlaceholder from '@/components/editor/PromptPlaceholder'

    // MANDATORY per CLAUDE.md: GrapesJS must always be dynamically imported with ssr: false
    const GrapesEditor = dynamic(() => import('@/components/editor/GrapesEditor'), { ssr: false })

    interface EditorClientWrapperProps {
      userEmail: string
    }

    export default function EditorClientWrapper({ userEmail }: EditorClientWrapperProps) {
      const editorRef = useRef<Editor | null>(null)

      function handleEditor(editor: Editor) {
        editorRef.current = editor
      }

      return (
        <div className="flex flex-col h-screen">
          <TopBar editorRef={editorRef} userEmail={userEmail} />
          <div className="flex-1 overflow-hidden">
            <GrapesEditor onEditor={handleEditor} />
          </div>
          <PromptPlaceholder />
        </div>
      )
    }
    ```

    Layout notes:
    - `h-screen` on the outer div makes the editor fill the full viewport height
    - `flex flex-col` stacks TopBar, canvas, and PromptPlaceholder vertically
    - The canvas `flex-1 overflow-hidden` takes all remaining height between TopBar and PromptPlaceholder
    - TopBar has a fixed height (from its own styling in Task 1 of 02-03)
    - PromptPlaceholder has its own height (from its own styling in Task 2 of 02-03)

    The dynamic import with `{ ssr: false }` satisfies the CLAUDE.md mandatory architecture rule. GrapesJS accesses `window` at module load time — without this, server-side rendering crashes.

    The `editorRef` ref object (not just its `.current` value) is passed to TopBar so TopBar can call `.current?.runCommand()` imperatively after the editor initializes. React refs are stable objects — passing the ref itself is correct (not `editorRef.current`).

    After creating, run `npm run build` to check for TypeScript errors.
